let lname = document.querySelector("#lastName");
lname.addEventListener('input', checkLastName);

function checkLastName(){
    if (lname.value.length>1){
        lname.classList.add("greenBox");
        lname.classList.remove("redBox");
        lname.nextElementSibling.classList.add('errorText')
         
        // console.log("true")
    } else {
        lname.classList.add("redBox");
        lname.classList.remove("greenBox");
        lname.nextElementSibling.classList.remove('errorText') 
        // console.log("false")
    }       
}

let age = document.querySelector("#age");
age.addEventListener("input", checkAge);

function checkAge(){
    if(age.value>=5 && age.value<=140){
        age.classList.add("greenBox");
        age.classList.remove("redBox");
        age.nextElementSibling.classList.add('errorText') 

    }else {
        age.classList.add("redBox");
        age.classList.remove("greenBox");
        age.nextElementSibling.classList.remove('errorText') 

    }
}
